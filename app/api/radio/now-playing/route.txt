import { NextResponse } from "next/server";
import { getAzuraNowPlayingUrl } from "@/lib/azuracast";

export const dynamic = "force-dynamic";

export async function GET() {
  try {
    const url = getAzuraNowPlayingUrl();

    const response = await fetch(url, {
      cache: "no-store",
      headers: {
        Accept: "application/json",
      },
    });

    const text = await response.text();

    if (!response.ok) {
      return NextResponse.json(
        {
          ok: false,
          error: "AzuraCast now-playing request failed.",
          status: response.status,
          details: text,
        },
        { status: response.status }
      );
    }

    const data = JSON.parse(text);

    const song = data?.now_playing?.song || {};
    const station = data?.station || {};
    const listeners = data?.listeners || {};

    return NextResponse.json({
      ok: true,
      raw: data,
      station: {
        name: station?.name || "Tha Core Online Radio",
        shortcode: station?.shortcode || null,
      },
      nowPlaying: {
        title: song?.title || "Unknown Title",
        artist: song?.artist || "Unknown Artist",
        text: song?.text || `${song?.artist || "Unknown Artist"} - ${song?.title || "Unknown Title"}`,
        album: song?.album || null,
        art: song?.art || null,
        elapsed: data?.now_playing?.elapsed || 0,
        duration: data?.now_playing?.duration || 0,
        isLive: data?.live?.is_live || false,
        streamer: data?.live?.streamer_name || null,
      },
      listeners: {
        current: listeners?.current || 0,
        unique: listeners?.unique || 0,
        total: listeners?.total || 0,
      },
    });
  } catch (error) {
    return NextResponse.json(
      {
        ok: false,
        error: error instanceof Error ? error.message : "Unknown now-playing error.",
      },
      { status: 500 }
    );
  }
}
